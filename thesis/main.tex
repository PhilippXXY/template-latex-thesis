% -----------------------------
% Glossary / abbreviations / symbols (bib2gls workflow)
% -----------------------------
% - record: enables the "record" indexing approach that bib2gls uses
% - abbreviations: creates a dedicated "abbreviations" glossary type
% - symbols: creates a dedicated "symbols" glossary type
\usepackage[record,abbreviations,symbols]{glossaries-extra}

% Pick how abbreviations appear in running text (first vs later use).
% Must be set before abbreviations are defined/loaded.
\setabbreviationstyle{long-short}

% Abbreviations: load into the abbreviations glossary type
\GlsXtrLoadResources[
  src={glossary/abbreviations},
  type=\glsxtrabbrvtype,
  selection=all,
  field-aliases={
    long_lowercase=long,
    long_titlecase=user1
  }
]

% Symbols: load into the symbols glossary type
\GlsXtrLoadResources[
  src={glossary/symbols},
  type=symbols,
  selection=all,
  field-aliases={
    symbol=name,
    unit=user1
  }
]
